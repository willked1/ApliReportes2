create table ciclista(
id number(5),
nombre varchar2(20),
apellido varchar2(20),
equipo varchar2(20),
edad number(2),
pais varchar2(20),
CONSTRAINT PK_CICLISTA PRIMARY KEY(id));


create table competencia(
id number(5),
nombre varchar2(20),
pais varchar2(20),
mes varchar2(20),
CONSTRAINT PK_COPETENCIA PRIMARY KEY(id));


create table VUELTA(
idciclista number(5),
idcompetencia number(5),
CONSTRAINT FK_CICLISTA FOREIGN KEY (idciclista) REFERENCES ciclista(id),
CONSTRAINT FK_COMPETENCIA FOREIGN KEY (idcompetencia) REFERENCES competencia(id));
